// دالة للتبديل بين التبويبات function switchTab(tabName) { // إخفاء جميع المحتويات document.querySelectorAll('.tab-content').forEach(tab => { tab.classList.remove('active'); }); // إزالة النشاط من جميع الأزرار document.querySelectorAll('.tab-button').forEach(button => { button.classList.remove('active'); }); // إظهار المحتوى المطلوب وتفعيل الزر if (tabName === 'square') { document.getElementById('squareContent').classList.add('active'); document.getElementById('squareTab').classList.add('active'); } else if (tabName === 'official') { document.getElementById('officialContent').classList.add('active'); document.getElementById('officialTab').classList.add('active'); } } // تشغيل الدالة عند تحميل الصفحة document.addEventListener('DOMContentLoaded', function() { createCurrencyList(); // إضافة مستمعي الأحداث للأزرار document.getElementById('squareTab').addEventListener('click', function() { switchTab('square'); }); document.getElementById('officialTab').addEventListener('click', function() { switchTab('official'); }); });
